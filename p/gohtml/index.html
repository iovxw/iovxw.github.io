еҹәжң¬е°ұжҳҜеҶҷзқҖзҺ©зҡ„

еҸҜд»Ҙе°ҶGolangе’ҢHTMLеҶҷеңЁдёҖиө·пјҢз„¶еҗҺз”Ёжң¬е·Ҙе…·иҪ¬жҚўжҲҗGolangж–Үд»¶

package tpl

import (
	"fmt"
	"bytes"
)

func Index() string {
	_buffer := new(bytes.Buffer)

	<a>Text</a>

	for i:=0; i<10; i++ {
		<a>{вҖӢ{string(i)}}</a>
	}

	var t string = "2"
	fmt.Println(t)

	print:=func() {
		<a>text</a>
	}
	print()

	test1(_buffer)
	test2(_buffer, t)

	return _buffer.String()
}

func test1(_buffer *bytes.Buffer) {
	<a>test1</a>
}

func test2(_buffer *bytes.Buffer, t string) {
	<a>test{вҖӢ{t}}</a>
}

иҪ¬жҚўеҗҺпјҡ

package tpl

import (
	"fmt"
	"bytes"
)

func Index() string {
	_buffer := new(bytes.Buffer)

	_buffer.WriteString("<a>Text</a>")

	for i:=0; i<10; i++ {
		_buffer.WriteString("<a>")
		_buffer.WriteString(string(i))
		_buffer.WriteString("</a>")
	}

	var t string = "2"
	fmt.Println(t)

	print:=func() {
		_buffer.WriteString("<a>text</a>")
	}
	print()

	test1(_buffer)
	test2(_buffer,t)

	return _buffer.String()
}

func test1(_buffer *bytes.Buffer) {
	_buffer.WriteString("<a>test1</a>")
}

func test2(_buffer *bytes.Buffer, t string) {
	_buffer.WriteString("<a>test")
	_buffer.WriteString(t)
	_buffer.WriteString("</a>")
}

е‘Ҫд»Өз»“жһ„:
  е‘Ҫд»Ө [еҸӮж•°] <жЁЎжқҝж–Үд»¶еӨ№и·Ҝеҫ„>

еҸӮж•°пјҡ
  -dl <еӯ—з¬ҰдёІ>     | й»ҳи®Өпјҡ{{      | е·ҰеҲҶйҡ”з¬Ұж ·ејҸ
  -dr <еӯ—з¬ҰдёІ>     | й»ҳи®Өпјҡ}}      | еҸіеҲҶйҡ”з¬Ұж ·ејҸ
  -suffix <еӯ—з¬ҰдёІ> | й»ҳи®Өпјҡgohtml  | GoHTMLжЁЎжқҝж–Үд»¶еҗҺзјҖ
  -buffer <еӯ—з¬ҰдёІ> | й»ҳи®Өпјҡ_buffer | зј“еҶІеҷЁеҸҳйҮҸеҗҚз§°

дёҫдҫӢпјҡ
  $ gohtml -dl <{ -dr }> -suffix temp -buffer buf /home/bluek404/gocode/web/view
   | еҲҷдјҡе°Ҷ
   | /home/bluek404/gocode/web/view
   | йҮҢйқўжүҖжңүtempдёәеҗҺзјҖзҡ„ж–Үд»¶иҪ¬жҚўдёәgoж–Үд»¶еҗҺж”ҫеҲ°еҗҢдёҖж–Үд»¶еӨ№еҶ…
   | е°Ҷзј“еҶІеҷЁеҸҳйҮҸеҗҚз§°и®ҫдёәвҖңbufвҖқ
   | е№¶и®ҫзҪ®е·ҰеҲҶйҡ”з¬ҰдёәвҖң<{вҖқ пјҢеҸіеҲҶеүІз¬ҰдёәвҖң}>вҖқ

еӨҮжіЁпјҡ
  ж–№жӢ¬еҸ·[]дёәйҖүеЎ«йЎ№зӣ®пјҢе°–жӢ¬еҸ·<>дёәеҝ…еЎ«йЎ№зӣ®

жәҗз Ғи§„иҢғ

дҪңдёәhttp.HandleFuncзҡ„еҮҪж•°пјҢеҝ…йЎ»иҝ”еӣһеҖјеҸӘжңүдёҖдёӘstringпјҲеӣ дёәеӨҡдәҶе°ұдёҚз¬ҰеҗҲhttp.HandleFuncзҡ„иҰҒжұӮдәҶпјү

з„¶еҗҺйҮҢйқўеҝ…йЎ»еҲӣе»әдёҖдёӘзұ»еһӢдёә*bytes.Bufferзҡ„еҸҳйҮҸпјҢеҸҳйҮҸеҗҚз§°йңҖиҰҒе’Ңи®ҫзҪ®зҡ„дёҖж ·пјҲй»ҳи®ӨжҳҜ_bufferпјүгҖӮиҝҷдёӘзј“еҶІеҷЁжҳҜз”ЁдәҺеӯҳж”ҫиҰҒиҫ“еҮәзҡ„HTMLд»Јз Ғзҡ„

з„¶еҗҺдҪңдёәhttp.HandleFuncзҡ„еҮҪж•°зҡ„иҝ”еӣһеҖјйңҖиҰҒжҳҜ_buffer.String()пјҢеҰӮжһңдҪ иҮӘе®ҡд№үдәҶзј“еҶІеҷЁеҸҳйҮҸеҗҚз§°пјҢйңҖиҰҒзӣёеә”зҡ„ж”№еҸҳ

иҙҹиҙЈеӨ„зҗҶйғЁеҲҶhtmlзҡ„еҮҪж•°пјҢжҺЁиҚҗеҶҷжҲҗдёҠйқўдҫӢеӯҗйҮҢзҡ„ж ·ејҸпјҲжҺҘж”¶дёҖдёӘзј“еҶІеҷЁеҸҳйҮҸпјүпјҢе°ұеғҸпјҡ

func test1(_buffer *bytes.Buffer) {
    <a>test1</a>
}

еӣ дёәдјҡиҪ¬жҚўжҲҗпјҡ

func test1(_buffer *bytes.Buffer) {
    _buffer.WriteString("<a>test1</a>")
}

иҝҷдёӘж ·еӯҗ

еҪ“з„¶дҪ иҰҒжҳҜеҶҷжҲҗеҮҪж•°йҮҢж–°е»әдёҖдёӘзј“еҶІеҷЁеҸҳйҮҸз„¶еҗҺеҶҚreturnд№ҹжҳҜжІЎй—®йўҳзҡ„

дёҚиҝҮйӮЈж ·е°ұйңҖиҰҒеңЁдҪҝз”ЁеҮҪж•°зҡ„ең°ж–№еҶҚеҠ дёҖдёӘиҫ“еҮәдәҶпјҲзӣёеҪ“йә»зғҰпјү

жүҖд»Ҙе°ұд№–д№–жҢүз…§ж ҮеҮҶеҶҷеҗ§

з„¶еҗҺиҝҳжңүдёҖзӮ№иҰҒжіЁж„Ҹ

е°ұжҳҜеҲҶйҡ”з¬Ұиҫ“еҮәеҸҳйҮҸйӮЈйҮҢпјҢжҜ”еҰӮ{вҖӢ{t}}

иҝҷйҮҢзҡ„еҲҶйҡ”з¬ҰеҸӘжңүеңЁHTMLдёӯжүҚдјҡиў«иҪ¬жҚўпјҢе°ұеғҸ<a>{вҖӢ{t}}</a>

еҶҷеңЁеҲ«зҡ„ең°ж–№жҳҜдёҚдјҡиҪ¬жҚўзҡ„пјҢжҜ”еҰӮпјҡ

<a>
if i==0 {
    {вҖӢ{t1}}
}else{
    {вҖӢ{t2}}
}
</a>

еҝ…йЎ»еҶҷжҲҗ

if i==0 {
    <a>{вҖӢ{t1}}</a>
}else{
    <a>{вҖӢ{t2}}</a>
}

иҝҷж ·жүҚеҸҜд»Ҙ

иҮӘеҠЁиҪ¬жҚў

жҜҸж¬Ўжӣҙж–°е®ҢжЁЎжқҝеҗҺйғҪиҰҒжүӢеҠЁжү§иЎҢдёҖж¬ЎиҪ¬жҚўе·Ҙе…·дёҖе®ҡжҳҜи¶…йә»зғҰзҡ„

жүҖд»ҘйңҖиҰҒиҮӘеҠЁзҡ„е·Ҙе…·

иҝҷйҮҢжҺЁиҚҗUnknwonзҡ„BRAпјҲеҗҚеӯ—еҘҪеғҸжңүзӮ№еҘҮжҖӘпјҹпјүпјҢеҪ“з„¶е…¶д»–зӣёдјјеҠҹиғҪзҡ„е·Ҙе…·д№ҹжҳҜеҸҜд»Ҙзҡ„пјҲеҰӮжһңжңүжӣҙеҘҪзҡ„иҝҳжӢңжүҳжҺЁиҚҗдёҖдёӢпјү
