  1. package main
  3. import (
  4.     "errors"
  5.     "fmt"
  6.     "github.com/go-gl/gl"
  7.     glfw "github.com/go-gl/glfw3"
  8.     "image"
  9.     "image/png"
  10.     "io"
  11.     "os"
  12. )
  14. const (
  15.     Title  = "Spinning Gopher"
  16.     Width  = 640
  17.     Height = 480
  18. )
  20. var (
  21.     texture    gl.Texture
  22.     rotx, roty float32
  23.     ambient    []float32 = []float32{0.5, 0.5, 0.5, 1}
  24.     diffuse    []float32 = []float32{1, 1, 1, 1}
  25.     lightpos   []float32 = []float32{-5, 5, 10, 0}
  26. )
  28. func errorCallback(err glfw.ErrorCode, desc string) {
  29.     fmt.Printf("%v: %v\n", err, desc)
  30. }
  32. func main() {
  33.     glfw.SetErrorCallback(errorCallback)
  35.     if !glfw.Init() {
  36.         panic("Can't init glfw!")
  37.     }
  38.     defer glfw.Terminate()
  40.     window, err := glfw.CreateWindow(Width, Height, Title, nil, nil)
  41.     if err != nil {
  42.         panic(err)
  43.     }
  45.     window.MakeContextCurrent()
  47.     glfw.SwapInterval(1)
  49.     gl.Init()
  51.     if err := initScene(); err != nil {
  52.         fmt.Fprintf(os.Stderr, "init: %s\n", err)
  53.         return
  54.     }
  55.     defer destroyScene()
  57.     for !window.ShouldClose() {
  58.         drawScene()
  59.         window.SwapBuffers()
  60.         glfw.PollEvents()
  61.     }
  62. }
  64. func createTexture(r io.Reader) (gl.Texture, error) {
  65.     img, err := png.Decode(r)
  66.     if err != nil {
  67.         return gl.Texture(0), err
  68.     }
  70.     rgbaImg, ok := img.(*image.NRGBA)
  71.     if !ok {
  72.         return gl.Texture(0), errors.New("texture must be an NRGBA image")
  73.     }
  75.     textureId := gl.GenTexture()
  76.     textureId.Bind(gl.TEXTURE_2D)
  77.     gl.TexParameterf(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR)
  78.     gl.TexParameterf(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR)
  80.     // flip image: first pixel is lower left corner
  81.     imgWidth, imgHeight := img.Bounds().Dx(), img.Bounds().Dy()
  82.     data := make([]byte, imgWidth*imgHeight*4)
  83.     lineLen := imgWidth * 4
  84.     dest := len(data) - lineLen
  85.     for src := 0; src < len(rgbaImg.Pix); src += rgbaImg.Stride {
  86.         copy(data[dest:dest+lineLen], rgbaImg.Pix[src:src+rgbaImg.Stride])
  87.         dest -= lineLen
  88.     }
  89.     gl.TexImage2D(gl.TEXTURE_2D, 0, 4, imgWidth, imgHeight, 0, gl.RGBA, gl.UNSIGNED_BYTE, data)
  91.     return textureId, nil
  92. }
  94. func initScene() (err error) {
  95.     gl.Enable(gl.TEXTURE_2D)
  96.     gl.Enable(gl.DEPTH_TEST)
  97.     gl.Enable(gl.LIGHTING)
  99.     gl.ClearColor(0.5, 0.5, 0.5, 0.0)
  100.     gl.ClearDepth(1)
  101.     gl.DepthFunc(gl.LEQUAL)
  103.     gl.Lightfv(gl.LIGHT0, gl.AMBIENT, ambient)
  104.     gl.Lightfv(gl.LIGHT0, gl.DIFFUSE, diffuse)
  105.     gl.Lightfv(gl.LIGHT0, gl.POSITION, lightpos)
  106.     gl.Enable(gl.LIGHT0)
  108.     gl.Viewport(0, 0, Width, Height)
  109.     gl.MatrixMode(gl.PROJECTION)
  110.     gl.LoadIdentity()
  111.     gl.Frustum(-1, 1, -1, 1, 1.0, 10.0)
  112.     gl.MatrixMode(gl.MODELVIEW)
  113.     gl.LoadIdentity()
  115.     goph, err := os.Open("gopher.png")
  116.     if err != nil {
  117.         panic(err)
  118.     }
  119.     defer goph.Close()
  121.     texture, err = createTexture(goph)
  122.     return
  123. }
  125. func destroyScene() {
  126.     texture.Delete()
  127. }
  129. func drawScene() {
  130.     gl.Clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT)
  132.     gl.MatrixMode(gl.MODELVIEW)
  133.     gl.LoadIdentity()
  134.     gl.Translatef(0, 0, -3.0)
  135.     gl.Rotatef(rotx, 1, 0, 0)
  136.     gl.Rotatef(roty, 0, 1, 0)
  138.     rotx += 0.5
  139.     roty += 0.1
  141.     texture.Bind(gl.TEXTURE_2D)
  143.     gl.Color4f(1, 1, 1, 1)
  145.     gl.Begin(gl.QUADS)
  147.     gl.Normal3f(0, 0, 1)
  148.     gl.TexCoord2f(0, 0)
  149.     gl.Vertex3f(-1, -1, 1)
  150.     gl.TexCoord2f(1, 0)
  151.     gl.Vertex3f(1, -1, 1)
  152.     gl.TexCoord2f(1, 1)
  153.     gl.Vertex3f(1, 1, 1)
  154.     gl.TexCoord2f(0, 1)
  155.     gl.Vertex3f(-1, 1, 1)
  157.     gl.Normal3f(0, 0, -1)
  158.     gl.TexCoord2f(1, 0)
  159.     gl.Vertex3f(-1, -1, -1)
  160.     gl.TexCoord2f(1, 1)
  161.     gl.Vertex3f(-1, 1, -1)
  162.     gl.TexCoord2f(0, 1)
  163.     gl.Vertex3f(1, 1, -1)
  164.     gl.TexCoord2f(0, 0)
  165.     gl.Vertex3f(1, -1, -1)
  167.     gl.Normal3f(0, 1, 0)
  168.     gl.TexCoord2f(0, 1)
  169.     gl.Vertex3f(-1, 1, -1)
  170.     gl.TexCoord2f(0, 0)

  172.     gl.TexCoord2f(1, 0)

  178.     gl.TexCoord2f(1, 1)
  179.     gl.Vertex3f(-1, -1, -1)
  180.     gl.TexCoord2f(0, 1)
  181.     gl.Vertex3f(1, -1, -1)
  182.     gl.TexCoord2f(0, 0)

  184.     gl.TexCoord2f(1, 0)
  185.     gl.Vertex3f(-1, -1, 1)

  188.     gl.TexCoord2f(1, 0)
  189.     gl.Vertex3f(1, -1, -1)
  190.     gl.TexCoord2f(1, 1)
  191.     gl.Vertex3f(1, 1, -1)

  200.     gl.TexCoord2f(1, 0)
  201.     gl.Vertex3f(-1, -1, 1)
  202.     gl.TexCoord2f(1, 1)
  203.     gl.Vertex3f(-1, 1, 1)
  204.     gl.TexCoord2f(0, 1)
  205.     gl.Vertex3f(-1, 1, -1)
  207.     gl.End()
  208. }

于是写了这么一个教程备用

sudo apt-get install libxrandr-dev libxi-dev

然后make+sudo make install

让XP继续接受更新

通过修改注册表来让XP继续接受更新

2014-05-27 00:00

在Betanews看到了让XP继续更新的方法

大体过程如下：

1.新建一个文件扩展名为.reg的文件（例如updates.reg）

2.用文本编辑器打开

3.修改内容为

Windows Registry Editor Version 5.00
[HKEY_LOCAL_MACHINE\SYSTEM\WPA\PosReady]
"Installed"=dword:00000001

4.保存，然后双击它

现在打开更新管理器就会发现有新的更新可用了

这个的原理是欺骗Windows Update，让它把你的电脑识别为Windows Embedded POSReady 2009

而Windows Embedded POSReady 2009的支持日期一直到2019年4月9日

当然这不是最好的解决办法，要想一直有免费升级的话，还是去用Linux吧

最起码你不用再担心系统是不是安全了

Front Matter For Fun

This is the meta data for this post. It is located at the top of the content/post/hello-hugo.md markdown file.

---
title: "Hello Hugo!"
description: "Saying 'Hello' from Hugo"
date: "2014-04-09"
categories:
  - "example"
  - "hello"
tags:
  - "example"
  - "hugo"
  - "blog"
---

This section, called 'Front Matter', is what tells Hugo about the content in this file. The title of the item,the description, the date it was posted. In our example we're adding two custom bits of data, too. The categories and tags sections are used in this example for indexing/grouping content.
